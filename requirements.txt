# --------------------------------------------------------------------------
# requirements.txt for gesture_control.py
#
# Before installing these, on Ubuntu/Debian you MUST install dlib's build
# tools first (dlib compiles from source and has no prebuilt wheel):
#
#   sudo apt install -y build-essential cmake libopenblas-dev liblapack-dev
#
# Then, inside a virtual environment:
#
#   python3 -m venv venv
#   source venv/bin/activate
#   pip install -r requirements.txt
#
# First run: look at the camera and press 'e' to enroll your face.

# --- Core CV / hand tracking ---
opencv-python
numpy<2

# The pipeline now uses the MediaPipe *Tasks* HandLandmarker (see
# hand_landmarker.py), not the deprecated mp.solutions API. That removes the
# reason for the old `mediapipe==0.10.14` pin: the mp.solutions packaging
# regression on Python 3.12 no longer affects us, because we don't import
# mp.solutions anywhere. Current mediapipe is fine.
#
# The Tasks API needs a model bundle downloaded once (not a pip dependency):
#   mkdir -p models
#   curl -L -o models/hand_landmarker.task \
#     https://storage.googleapis.com/mediapipe-models/hand_landmarker/hand_landmarker/float16/1/hand_landmarker.task
mediapipe>=0.10.14
protobuf>=4.25.3,<5

# --- Face recognition: REMOVED ---
# face_recognition / dlib / setuptools<82 used to live here for the face-based
# operator lock. That lock is now pose-based (see gesture_control.py), so the
# whole dlib dependency is gone - along with its from-source compile, its
# pkg_resources/setuptools pin, its CPU cost as the pipeline's slowest step,
# and the problem of having no aarch64 build path onto the Jetson.
#
# The old implementation is preserved on the `face-identity-lock` branch, which
# still needs those packages if you check it out.

# --- Gesture classifier (training + inference) ---
# RandomForest over engineered motion features. Chosen over an LSTM because
# it trains in seconds, adds no measurable latency to the live loop, and
# works well with the ~30-80 samples/class you can realistically record.
# joblib ships with scikit-learn but is pinned explicitly since we call it
# directly to save/load the model bundle.
scikit-learn
joblib
